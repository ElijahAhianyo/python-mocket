def do_the_magic(lib_magic, body):
    if hasattr(lib_magic, 'from_buffer'):
        return lib_magic.from_buffer(body, mime=True)
    _magic = lib_magic.open(lib_magic.MAGIC_MIME_TYPE)
    _magic.load()
    return _magic.buffer(body)
